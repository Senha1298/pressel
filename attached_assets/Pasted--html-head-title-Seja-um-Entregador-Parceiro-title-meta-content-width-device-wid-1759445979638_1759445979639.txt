  <script>
    // Modal open/close
    const openModalBtn = document.getElementById('openModalBtn');
    const modalCadastro = document.getElementById('modalCadastro');
    const closeModalBtn = document.getElementById('closeModalBtn');
    const cepInput = document.getElementById('cepInput');
    const continuarBtn = document.getElementById('continuarBtn');
    const cepForm = document.getElementById('cepForm');

    openModalBtn.addEventListener('click', function() {
      modalCadastro.classList.remove('hidden');
      setTimeout(() => {
        cepInput.focus();
      }, 100);
    });

    closeModalBtn.addEventListener('click', function() {
      modalCadastro.classList.add('hidden');
      cepInput.value = '';
      continuarBtn.disabled = true;
    });

    // Fechar modal ao clicar fora
    modalCadastro.addEventListener('mousedown', function(e) {
      if (e.target === modalCadastro) {
        modalCadastro.classList.add('hidden');
        cepInput.value = '';
        continuarBtn.disabled = true;
      }
    });

    // CEP mask and validation
    function formatCep(value) {
      value = value.replace(/\D/g, '');
      if (value.length > 5) {
        value = value.slice(0,5) + '-' + value.slice(5,8);
      }
      return value.slice(0,9);
    }

    function isValidCep(value) {
      return /^\d{5}-\d{3}$/.test(value);
    }

    cepInput.addEventListener('input', function(e) {
      let value = e.target.value;
      value = formatCep(value);
      e.target.value = value;
      continuarBtn.disabled = !isValidCep(value);
    });

    // Prevent non-numeric input
    cepInput.addEventListener('keypress', function(e) {
      if (!/[0-9]/.test(e.key)) {
        e.preventDefault();
      }
    });

    // Prevent form submit (for demo)
    cepForm.addEventListener('submit', function(e) {
      e.preventDefault();
      // Aqui você pode adicionar a lógica de envio do CEP
      modalCadastro.classList.add('hidden');
      cepInput.value = '';
      continuarBtn.disabled = true;
      alert('CEP enviado com sucesso!');
    });
  </script>
